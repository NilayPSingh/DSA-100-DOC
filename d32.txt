#include <stdio.h>
#include <stdlib.h>

#define MAX 1000

int stack[MAX], top = -1;

void push(int val) {
    stack[++top] = val;
}

void pop() {
    if (top != -1) top--;
}

void display() {
    for (int i = top; i >= 0; i--) printf("%d ", stack[i]);
    if (top != -1) printf("\n");
}

int main() {
    int n, m, val;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &val);
        push(val);
    }
    scanf("%d", &m);
    for (int i = 0; i < m; i++) pop();
    display();
    return 0;
}
