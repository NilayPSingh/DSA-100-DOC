#include <stdio.h>
#include <stdlib.h>

struct Node {
    int coef;
    int exp;
    struct Node* next;
};

struct Node* append(struct Node* head, int coef, int exp) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->coef = coef;
    newNode->exp = exp;
    newNode->next = NULL;
    if (!head) return newNode;
    struct Node* temp = head;
    while (temp->next) temp = temp->next;
    temp->next = newNode;
    return head;
}

void printPoly(struct Node* head) {
    while (head) {
        if (head->exp > 1) printf("%dx^%d", head->coef, head->exp);
        else if (head->exp == 1) printf("%dx", head->coef);
        else printf("%d", head->coef);
        if (head->next) printf(" + ");
        head = head->next;
    }
    printf("\n");
}

int main() {
    int n, c, e;
    struct Node* head = NULL;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%d %d", &c, &e);
        head = append(head, c, e);
    }
    printPoly(head);
    return 0;
}
